Widok ze wszystkimi zawodnikami (posortowany wzgledem zbiorek):
CREATE VIEW rebounds_view AS
SELECT p.first_name || ' ' || p.last_name AS 'Player', c.city_name || ' ' || c.club_name AS 'Club', s.points AS 'Points', s.assists AS 'Assists', s.rebounds AS 'Rebounds'
FROM players AS p
INNER JOIN stats AS s
ON p.player_id = s.player_id
INNER JOIN clubs AS c
ON c.club_id = p.club_id
ORDER BY s.rebounds DESC;
